# ダイクストラ



```cpp


vector<int> ds(vector<vector<pair<int,int>>>& v,int str){
    int inf = 1e9;
    vector<int> ans(v.size(),inf);
    ans[str] = 0;
    priority_queue<pair<int,int>,vector<pair<int,int>>,greater<pair<int,int>>> q;
    q.push({0,str});
    while(q.size()!=0){
        int cos = q.top().first;
        int pos = q.top().second;
        q.pop();
        if(cos > ans[pos])continue;
        for(pair<int,int> i : v[pos]){
            if(ans[i.first] > cos + i.second){
                ans[i.first] = cos + i.second;
                q.push({cos+i.second,i.first});
            }
        }
    }
    return ans;
}
```
